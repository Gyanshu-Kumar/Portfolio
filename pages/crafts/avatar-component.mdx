    # Get started with the type and interface

    ```ts
    type AvatarSizeType = 'small' | 'medium' | 'large' | 'extraLarge';
    ```

    # Adding utility `cn()` method

        Create a file `lib/utils.ts` and install packages: `clsx` and `tailwind-merge`

        ## Installing Dependencies

        ```cmd
        npm install clsx tailwind-merge --save-dev
        ```

        Then add the code below the the `lib/utils` file,

        ```tsx
        import { ClassValue, clsx } from 'clsx';
        import { twMerge } from 'tailwind-merge';

        export function cn(...inputs: ClassValue[]) {
            return twMerge(clsx(inputs));
        }
        ```

    # Create Function Component for Button

    ```tsx
    const Avatar: React.FunctionComponent<AvatarProps> = ({
      imgSrc,
      fallback,
      size = 'medium',
      className,
      ...attr
    }) => {
      return (
        <div
          className={cn(
            'avatar-content-wrapper rounded-full overflow-hidden',
            AvatarSizeClass[size],
            fallback
              ? 'bg-orange-500 text-white flex flex-row items-center justify-center cursor-default select-none'
              : '',
            className,
          )}
          {...attr}>
          {imgSrc &&
            typeof imgSrc === 'string' &&
            ((size === 'small' && (
              <Image src={imgSrc} alt={'avatar'} width={'32'} height={'32'} />
            )) ||
              (size === 'medium' && (
                <Image src={imgSrc} alt={'avatar'} width={'32'} height={'32'} />
              )) ||
              (size === 'large' && (
                <Image src={imgSrc} alt={'avatar'} width={'64'} height={'64'} />
              )) ||
              (size === 'extraLarge' && (
                <Image src={imgSrc} alt={'avatar'} width={'64'} height={'64'} />
              )))}
          {!imgSrc && fallback && (
            <>
              <span>{fallback}</span>
              <span className="sr-only">{fallback}</span>
            </>
          )}
        </div>
      );
    };

    export { Avatar };
    ```
